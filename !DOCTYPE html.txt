<script>
    const WORDS = ["SAGAZ", "NOBRE", "ALGOZ", "IDEIA", "PODER", "MUITO", "VIGOR", "HONRA", "JUSTO", "ETICO", "FORTE", "MUNDO", "TEMPO", "LIVRO", "SAUDE", "GENTE", "TERMO", "SONHO", "AREIA", "PORTA"];
    const SECRET = WORDS[Math.floor(Math.random() * WORDS.length)].toUpperCase();
    
    let currentAttempt = 0;
    let focusedIndex = 0; // Onde a letra vai cair
    let guesses = ["", "", "", "", "", ""]; 
    let gameOver = false;

    // Iniciar Tabuleiro
    const board = document.getElementById('game-board');
    for (let i = 0; i < 6; i++) {
        const row = document.createElement('div');
        row.className = 'row'; row.id = `r-${i}`;
        for (let j = 0; j < 5; j++) {
            const tile = document.createElement('div');
            tile.className = 'tile';
            tile.id = `t-${i}-${j}`;
            // Clique para focar
            tile.onclick = () => {
                if (i === currentAttempt && !gameOver) {
                    focusedIndex = j;
                    updateVisuals();
                }
            };
            row.appendChild(tile);
        }
        board.appendChild(row);
    }

    // Teclado
    const layout = [ "QWERTYUIOP".split(""), "ASDFGHJKL".split(""), ["ENTER", ..."ZXCVBNM".split(""), "BACK"] ];
    layout.forEach((row, i) => {
        const rowEl = document.getElementById(`row${i+1}`);
        row.forEach(k => {
            const b = document.createElement('button'); b.textContent = k === "BACK" ? "←" : k;
            b.onclick = () => handleInput(k); rowEl.appendChild(b);
        });
    });

    function handleInput(k) {
        if (gameOver) return;
        
        let currentWord = guesses[currentAttempt].split("");
        // Garantir que o array tenha 5 espaços
        while(currentWord.length < 5) currentWord.push("");

        if (k === "ENTER") {
            if (guesses[currentAttempt].length < 5) {
                document.getElementById(`r-${currentAttempt}`).classList.add('shake');
                setTimeout(() => document.getElementById(`r-${currentAttempt}`).classList.remove('shake'), 400);
            } else {
                submit();
            }
        } else if (k === "BACK" || k === "BACKSPACE") {
            // Se a caixa atual estiver vazia, volta uma e apaga
            if (currentWord[focusedIndex] === "" && focusedIndex > 0) {
                focusedIndex--;
            }
            currentWord[focusedIndex] = "";
            guesses[currentAttempt] = currentWord.join("");
            updateVisuals();
        } else if (/^[A-Z]$/i.test(k)) {
            currentWord[focusedIndex] = k.toUpperCase();
            guesses[currentAttempt] = currentWord.join("");
            // Pula para a próxima caixa após digitar
            if (focusedIndex < 4) focusedIndex++;
            updateVisuals();
        }
    }

    function updateVisuals() {
        const word = guesses[currentAttempt];
        for (let j = 0; j < 5; j++) {
            const t = document.getElementById(`t-${currentAttempt}-${j}`);
            t.textContent = word[j] || "";
            t.classList.remove('focused', 'filled');
            if (j === focusedIndex && !gameOver) t.classList.add('focused');
            if (word[j]) t.classList.add('filled');
            t.classList.add('active-row');
        }
    }

    function submit() {
        const guess = guesses[currentAttempt];
        const secretArr = SECRET.split("");
        const results = new Array(5).fill("absent");

        for (let i = 0; i < 5; i++) {
            if (guess[i] === secretArr[i]) {
                results[i] = "correct";
                secretArr[i] = null;
            }
        }
        for (let i = 0; i < 5; i++) {
            if (results[i] === "correct") continue;
            let idx = secretArr.indexOf(guess[i]);
            if (idx !== -1) {
                results[i] = "present";
                secretArr[idx] = null;
            }
        }

        results.forEach((status, i) => {
            const t = document.getElementById(`t-${currentAttempt}-${i}`);
            t.classList.remove('focused');
            setTimeout(() => {
                t.classList.add(status);
                if (i === 4 && guess === SECRET) victory();
            }, i * 150);
        });

        if (guess === SECRET) {
            gameOver = true;
        } else if (currentAttempt === 5) {
            gameOver = true;
            setTimeout(() => showStats(false), 1500);
        } else {
            currentAttempt++;
            focusedIndex = 0;
            updateVisuals();
        }
    }

    function victory() {
        for (let i = 0; i < 5; i++) {
            setTimeout(() => document.getElementById(`t-${currentAttempt}-${i}`).classList.add('jump'), i * 100);
        }
        setTimeout(() => showStats(true), 1500);
    }

    function showStats(win) {
        const modal = document.getElementById('modal');
        document.getElementById('stats-text').innerHTML = win ? "🎉 BRILHANTE!" : `A palavra era: <b>${SECRET}</b>`;
        modal.style.display = "flex";
    }

    // Inicializar visual
    updateVisuals();
    window.onkeydown = (e) => handleInput(e.key.toUpperCase());
    
    // Tema
    const themeBtn = document.getElementById('theme-btn');
    themeBtn.onclick = () => {
        let t = document.documentElement.getAttribute('data-theme') === 'dark' ? 'light' : 'dark';
        document.documentElement.setAttribute('data-theme', t);
    };
</script>
